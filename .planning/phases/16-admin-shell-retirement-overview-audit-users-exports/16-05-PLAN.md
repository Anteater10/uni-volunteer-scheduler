    import { useState, useMemo } from "react";
    import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
    import api from "../lib/api";
    import { useAdminPageTitle } from "./admin/AdminLayout";
    import SideDrawer from "../components/admin/SideDrawer";
    import RoleBadge from "../components/admin/RoleBadge";
    import Button from "../components/ui/Button";
    import Input from "../components/ui/Input";
    import Label from "../components/ui/Label";
    import Modal from "../components/ui/Modal";
    import EmptyState from "../components/ui/EmptyState";
    import Skeleton from "../components/ui/Skeleton";
    import { useAuth } from "../lib/auth"; // or however current user is read — verify during execution

    const ROLES = ["admin", "organizer"]; // D-13: no participant

    const RTF = new Intl.RelativeTimeFormat("en", { numeric: "auto" });
    function relativeTime(iso) { /* same helper pattern */ }
    function lastLoginLabel(iso) { return iso ? relativeTime(iso) : "Never"; }

    export default function UsersAdminPage() {
      useAdminPageTitle("Users");
      const { currentUser } = useAuth(); // or equivalent — REPLACE with actual hook
      const qc = useQueryClient();

      // SEPARATE error states (fixes D-43.1 shared-err bug)
      const [createError, setCreateError] = useState(null);
      const [updateError, setUpdateError] = useState(null);

      const [showDeactivated, setShowDeactivated] = useState(false);
      const [search, setSearch] = useState("");
      const [roleFilter, setRoleFilter] = useState("all");
      const [drawerUser, setDrawerUser] = useState(null);
      const [inviteOpen, setInviteOpen] = useState(false);
      const [ccpaModal, setCcpaModal] = useState(null); // {type: "export"|"delete", user}

      const listQ = useQuery({
        queryKey: ["adminUsers", { include_inactive: showDeactivated }],
        queryFn: () => api.admin.users.list({ include_inactive: showDeactivated }),
      });

      const inviteM = useMutation({
        mutationFn: (body) => api.admin.users.invite(body),
        onSuccess: () => { qc.invalidateQueries(["adminUsers"]); setInviteOpen(false); setCreateError(null); },
        onError: (e) => setCreateError(e.message),
      });

      const updateM = useMutation({
        mutationFn: ({ id, patch }) => api.admin.users.update(id, patch),  // verify method name against api.js
        onSuccess: () => { qc.invalidateQueries(["adminUsers"]); setDrawerUser(null); setUpdateError(null); },
        onError: (e) => setUpdateError(e.message),
      });

      const deactivateM = useMutation({
        mutationFn: (id) => api.admin.users.deactivate(id),
        onSuccess: () => { qc.invalidateQueries(["adminUsers"]); setDrawerUser(null); },
        onError: (e) => setUpdateError(e.message),
      });

      const reactivateM = useMutation({
        mutationFn: (id) => api.admin.users.reactivate(id),
        onSuccess: () => qc.invalidateQueries(["adminUsers"]),
      });

      const users = listQ.data?.items || listQ.data || [];
      const filtered = useMemo(() => users.filter(u => {
        if (roleFilter !== "all" && u.role !== roleFilter) return false;
        const q = search.toLowerCase().trim();
        if (!q) return true;
        return (u.name || "").toLowerCase().includes(q) || (u.email || "").toLowerCase().includes(q);
      }), [users, search, roleFilter]);

      const activeAdminCount = users.filter(u => u.role === "admin" && u.is_active).length;

      // LOAD error ONLY — NOT createError (D-43.1 regression fix)
      if (listQ.isPending) return <Skeleton rows={5}/>;
      if (listQ.error) return <EmptyState title="Couldn't load users" body={listQ.error.message} action={<Button onClick={()=>listQ.refetch()}>Retry</Button>}/>;

      return (
        <div className="space-y-4">
          <div className="flex justify-between items-start">
            <div>
              <h1 className="text-2xl font-bold">Users</h1>
              <p className="text-gray-600">
                People who can sign into this admin panel. Invite organizers and other admins — students don't have accounts.
              </p>
            </div>
            <Button onClick={() => setInviteOpen(true)}>Invite user</Button>
          </div>

          <div className="flex gap-3 items-center">
            <Input placeholder="Search name or email..." value={search} onChange={e=>setSearch(e.target.value)}/>
            <select value={roleFilter} onChange={e=>setRoleFilter(e.target.value)}>
              <option value="all">All roles</option>
              <option value="admin">Admin</option>
              <option value="organizer">Organizer</option>
            </select>
            <label className="flex items-center gap-2">
              <input type="checkbox" checked={showDeactivated} onChange={e=>setShowDeactivated(e.target.checked)}/>
              Show deactivated
            </label>
          </div>

          <table className="w-full">
            <thead>
              <tr>
                <th className="text-left">Name</th>
                <th className="text-left">Email</th>
                <th className="text-left">Role</th>
                <th className="text-left">Last login</th>
                <th className="text-left">Status</th>
              </tr>
            </thead>
            <tbody>
              {filtered.map(u => (
                <tr
                  key={u.id}
                  className={`border-t cursor-pointer hover:bg-gray-50 ${!u.is_active ? "opacity-50" : ""}`}
                  onClick={() => { setDrawerUser(u); setUpdateError(null); }}
                >
                  <td>{u.name || "(no name)"}</td>
                  <td>{u.email}</td>
                  <td><RoleBadge role={u.role}/></td>
                  <td>{lastLoginLabel(u.last_login_at)}</td>
                  <td>
                    <span className={`px-2 py-1 rounded text-xs ${u.is_active ? "bg-green-100 text-green-800" : "bg-gray-100 text-gray-700"}`}>
                      {u.is_active ? "Active" : "Deactivated"}
                    </span>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>

          {/* Invite drawer */}
          <SideDrawer open={inviteOpen} onClose={() => { setInviteOpen(false); setCreateError(null); }} title="Invite user">
            <InviteForm
              onSubmit={(body) => inviteM.mutate(body)}
              submitting={inviteM.isPending}
              error={createError}
            />
          </SideDrawer>

          {/* Edit drawer */}
          <SideDrawer open={!!drawerUser} onClose={() => setDrawerUser(null)} title="Edit user">
            {drawerUser && (
              <EditUserForm
                user={drawerUser}
                currentUser={currentUser}
                activeAdminCount={activeAdminCount}
                onSave={(patch) => updateM.mutate({id: drawerUser.id, patch})}
                onDeactivate={() => deactivateM.mutate(drawerUser.id)}
                onReactivate={() => reactivateM.mutate(drawerUser.id)}
                onCcpaExport={() => setCcpaModal({type: "export", user: drawerUser})}
                onCcpaDelete={() => setCcpaModal({type: "delete", user: drawerUser})}
                error={updateError}
                saving={updateM.isPending || deactivateM.isPending}
              />
            )}
          </SideDrawer>

          {/* CCPA modals */}
          {ccpaModal?.type === "export" && (
            <Modal open onClose={() => setCcpaModal(null)} title="Export personal data">
              <p>Download all personal data we have on {ccpaModal.user.name || ccpaModal.user.email}. This includes their signups, check-ins, and audit trail entries. The download is a JSON file.</p>
              <Button onClick={async () => { await api.admin.users.ccpaExport(ccpaModal.user.id); setCcpaModal(null); }}>Download</Button>
            </Modal>
          )}
          {ccpaModal?.type === "delete" && (
            <CcpaDeleteModal user={ccpaModal.user} onClose={() => setCcpaModal(null)}/>
          )}
        </div>
      );
    }

    function InviteForm({ onSubmit, submitting, error }) {
      const [name, setName] = useState("");
      const [email, setEmail] = useState("");
      const [role, setRole] = useState("organizer");
      return (
        <form onSubmit={e => { e.preventDefault(); onSubmit({name, email, role}); }} className="space-y-3">
          <div><Label>Name</Label><Input value={name} onChange={e=>setName(e.target.value)} required/></div>
          <div><Label>Email</Label><Input type="email" value={email} onChange={e=>setEmail(e.target.value)} required/></div>
          <div>
            <Label>Role</Label>
            <select value={role} onChange={e=>setRole(e.target.value)}>
              <option value="organizer">Organizer</option>
              <option value="admin">Admin</option>
            </select>
          </div>
          {error && <p className="text-red-600">{error}</p>}
          <p className="text-xs text-gray-500">They'll get an email with a sign-in link. The link expires in 15 minutes.</p>
          <Button type="submit" disabled={submitting}>{submitting ? "Sending..." : "Send invite"}</Button>
        </form>
      );
    }

    function EditUserForm({ user, currentUser, activeAdminCount, onSave, onDeactivate, onReactivate, onCcpaExport, onCcpaDelete, error, saving }) {
      const [name, setName] = useState(user.name || "");
      const [role, setRole] = useState(user.role);
      const [universityId, setUniversityId] = useState(user.university_id || "");
      const [notifyEmail, setNotifyEmail] = useState(user.notify_email ?? true);

      const isSelf = currentUser?.id === user.id;
      const isLastAdmin = user.role === "admin" && activeAdminCount <= 1 && user.is_active;
      const cannotDeactivate = isSelf ? "You cannot deactivate your own account" : isLastAdmin ? "Cannot deactivate the last active admin" : null;
      const cannotDemote = isSelf && user.role === "admin" ? "You cannot demote your own admin account" : isLastAdmin ? "Cannot demote the last active admin" : null;

      return (
        <form onSubmit={e => { e.preventDefault(); onSave({name, role, university_id: universityId, notify_email: notifyEmail}); }} className="space-y-3">
          <div><Label>Name</Label><Input value={name} onChange={e=>setName(e.target.value)}/></div>
          <div>
            <Label>Email</Label>
            <Input value={user.email} disabled/>
            <p className="text-xs text-gray-500">Email can't be changed — delete and reinvite if needed.</p>
          </div>
          <div>
            <Label>Role</Label>
            <select value={role} onChange={e=>setRole(e.target.value)} disabled={!!cannotDemote && role === "admin"}>
              <option value="admin">Admin</option>
              <option value="organizer" disabled={!!cannotDemote}>Organizer{cannotDemote ? ` (${cannotDemote})` : ""}</option>
            </select>
          </div>
          <div><Label>University ID</Label><Input value={universityId} onChange={e=>setUniversityId(e.target.value)}/></div>
          <label className="flex items-center gap-2">
            <input type="checkbox" checked={notifyEmail} onChange={e=>setNotifyEmail(e.target.checked)}/>
            Email notifications on
          </label>

          {error && <p className="text-red-600">{error}</p>}

          <div className="flex gap-2">
            <Button type="submit" disabled={saving}>Save</Button>
            {user.is_active ? (
              <Button type="button" variant="danger" onClick={onDeactivate} disabled={!!cannotDeactivate} title={cannotDeactivate || ""}>
                Deactivate
              </Button>
            ) : (
              <Button type="button" onClick={onReactivate}>Reactivate</Button>
            )}
          </div>

          <hr/>
          <div className="space-y-2">
            <p className="text-sm font-semibold">CCPA (California privacy rights)</p>
            <Button type="button" variant="outline" onClick={onCcpaExport}>CCPA Data Export</Button>
            <Button type="button" variant="outline" onClick={onCcpaDelete}>CCPA Delete Account</Button>
          </div>
        </form>
      );
    }

    function CcpaDeleteModal({ user, onClose }) {
      const [confirm, setConfirm] = useState("");
      return (
        <Modal open onClose={onClose} title="Permanently delete personal data">
          <p>Permanently anonymize {user.name || user.email}'s account. Their signups and check-ins stay (for audit), but their name, email, and phone are replaced with "Anonymized user". This cannot be undone.</p>
          <Label>Type the user's email to confirm:</Label>
          <Input value={confirm} onChange={e=>setConfirm(e.target.value)}/>
          <Button variant="danger" disabled={confirm !== user.email} onClick={async () => { await api.admin.users.ccpaDelete(user.id); onClose(); }}>
            Delete account
          </Button>
        </Modal>
      );
    }
