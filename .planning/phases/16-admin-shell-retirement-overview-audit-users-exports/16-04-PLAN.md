    ```jsx
    function relativeTime(isoString) {
      const diff = (new Date(isoString) - new Date()) / 1000;
      const abs = Math.abs(diff);
      if (abs < 60) return RTF.format(Math.round(diff), "second");
      if (abs < 3600) return RTF.format(Math.round(diff / 60), "minute");
      if (abs < 86400) return RTF.format(Math.round(diff / 3600), "hour");
    ```

    ```jsx
    import { useSearchParams, Link } from "react-router-dom";
    import { useQuery } from "@tanstack/react-query";
    import api, { downloadBlob } from "../lib/api";
    import SideDrawer from "../components/admin/SideDrawer";
    import DatePresetPicker from "../components/admin/DatePresetPicker";
    import Pagination from "../components/admin/Pagination";
    import RoleBadge from "../components/admin/RoleBadge";
    import { useAdminPageTitle } from "./admin/AdminLayout";

    const PAGE_SIZE = 25;
    const RTF = new Intl.RelativeTimeFormat("en", { numeric: "auto" });
    function relativeTime(iso) { /* same helper as Overview */ }

    export default function AuditLogsPage() {
      useAdminPageTitle("Audit Logs");
      const [searchParams, setSearchParams] = useSearchParams();
      const page = parseInt(searchParams.get("page") || "1", 10);
      const q = searchParams.get("q") || "";
      const kind = searchParams.get("kind") || "";
      const actorId = searchParams.get("actor_id") || "";
      const fromDate = searchParams.get("from_date") || "";
      const toDate = searchParams.get("to_date") || "";

      const [selected, setSelected] = useState(null);

      const params = { page, page_size: PAGE_SIZE, q, kind, actor_id: actorId, from_date: fromDate, to_date: toDate };
      const q_data = useQuery({
        queryKey: ["auditLogs", params],
        queryFn: () => api.admin.auditLogs(params),
        keepPreviousData: true,
      });

      const actorsQ = useQuery({queryKey:["adminUsersForFilter"], queryFn: () => api.admin.users.list({include_inactive: true})});

      function updateParam(name, value) {
        const next = new URLSearchParams(searchParams);
        if (value === "" || value === null || value === undefined) next.delete(name);
        else next.set(name, value);
        next.set("page", "1");
        setSearchParams(next);
      }
      function setPage(p) {
        const next = new URLSearchParams(searchParams);
        next.set("page", String(p));
        setSearchParams(next);
      }

      const rows = q_data.data?.items || [];
      const total = q_data.data?.total || 0;
      const totalPages = Math.max(1, Math.ceil(total / PAGE_SIZE));

      return (
        <div className="space-y-4">
          <h1 className="text-2xl font-bold">Audit logs</h1>
          <p className="text-gray-600">
            This page shows a history of every important change to the system — who did what, when, and to what. Use the filters to narrow down what you're looking for.
          </p>

          <div className="flex flex-wrap gap-3 items-end border rounded p-3">
            <Input placeholder="Search..." value={q} onChange={e=>updateParam("q", e.target.value)}/>
            <Select value={kind} onChange={e=>updateParam("kind", e.target.value)}>
              <option value="">All actions</option>
              {/* Options fed from ACTION_LABELS; hard-code the same list mirror here */}
              <option value="signup_cancelled">Cancelled a signup</option>
              <option value="user_invite">Invited a new user</option>
              {/* ... more ... */}
            </Select>
            <Select value={actorId} onChange={e=>updateParam("actor_id", e.target.value)}>
              <option value="">All actors</option>
              {(actorsQ.data?.items || actorsQ.data || []).map(u => (
                <option key={u.id} value={u.id}>{u.name || u.email}</option>
              ))}
            </Select>
            <DatePresetPicker
              value={{preset: searchParams.get("preset") || "30d", from: fromDate, to: toDate}}
              onChange={({preset, from, to}) => {
                const next = new URLSearchParams(searchParams);
                next.set("preset", preset);
                if (from) next.set("from_date", from); else next.delete("from_date");
                if (to) next.set("to_date", to); else next.delete("to_date");
                next.set("page","1");
                setSearchParams(next);
              }}
              presets={["24h","7d","30d","quarter","custom"]}
            />
            <Button onClick={() => downloadBlob("/admin/audit-logs.csv", "audit-logs.csv", {params})}>
              Export filtered view (CSV)
            </Button>
          </div>

          {q_data.isPending ? <Skeleton/> : q_data.error ? <EmptyState title="Couldn't load audit logs" body={q_data.error.message}/> : (
            <table className="w-full">
              <thead>
                <tr>
                  <th className="text-left">When</th>
                  <th className="text-left">Who</th>
                  <th className="text-left">What</th>
                  <th className="text-left">Target</th>
                  <th className="text-left">Details</th>
                </tr>
              </thead>
              <tbody>
                {rows.map(log => (
                  <tr key={log.id} className="border-t hover:bg-gray-50 cursor-pointer" onClick={() => setSelected(log)}>
                    <td title={log.timestamp}>{relativeTime(log.timestamp)}</td>
                    <td><span className="font-medium">{log.actor_label}</span> <RoleBadge role={log.actor_role}/></td>
                    <td>{log.action_label}</td>
                    <td className="text-gray-700">{log.entity_label}</td>
                    <td><Button size="sm" onClick={(e) => {e.stopPropagation(); setSelected(log);}}>View</Button></td>
                  </tr>
                ))}
              </tbody>
            </table>
          )}

          <Pagination page={page} totalPages={totalPages} onChange={setPage}/>

          <SideDrawer open={!!selected} onClose={() => setSelected(null)} title="Audit log entry">
            {selected && (
              <div className="space-y-3">
                <div><strong>When:</strong> {selected.timestamp}</div>
                <div><strong>Who:</strong> {selected.actor_label} ({selected.actor_role})</div>
                <div><strong>What:</strong> {selected.action_label}</div>
                <div><strong>Target:</strong> {selected.entity_label}</div>
                <div>
                  <strong>Raw payload:</strong>
                  <pre className="bg-gray-50 p-3 rounded overflow-x-auto text-xs">{JSON.stringify(selected, null, 2)}</pre>
                  <Button onClick={() => navigator.clipboard.writeText(JSON.stringify(selected, null, 2))}>
                    Copy to clipboard
                  </Button>
                </div>
              </div>
            )}
          </SideDrawer>
        </div>
      );
    }
    ```
