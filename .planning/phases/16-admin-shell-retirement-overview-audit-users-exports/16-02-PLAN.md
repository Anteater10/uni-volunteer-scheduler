    ```python
    def invite_user(
        body: schemas.UserInvite,
        db: Session = Depends(get_db),

    ):

        )
        db.add(user)

        audit.log_action(db, actor_id=actor.id, action="user_invite",
                         entity_type="User", entity_id=user.id)
        db.commit()
        db.refresh(user)
        try:
            send_invite_email(user, db)
        except Exception as e:
            logger.error("invite email send failed", exc_info=e)
        return user

    def _count_active_admins_locked(db: Session, exclude_id=None) -> int:
        q = db.query(models.User).filter(
            models.User.role == models.UserRole.admin,
            models.User.is_active == True,  # noqa: E712
            models.User.deleted_at.is_(None),
        ).with_for_update()
        if exclude_id is not None:
            q = q.filter(models.User.id != exclude_id)
        return q.count()

    @router.post("/{user_id}/deactivate", response_model=schemas.UserRead)
    def deactivate_user(
        user_id: UUID,
        db: Session = Depends(get_db),
        actor: models.User = Depends(require_role(models.UserRole.admin)),
    ):
        if actor.id == user_id:
            raise HTTPException(409, "You cannot deactivate your own account")
        user = db.query(models.User).filter(models.User.id == user_id).with_for_update().first()
        if not user:
            raise HTTPException(404, "User not found")
        if user.role == models.UserRole.admin:
            remaining = _count_active_admins_locked(db, exclude_id=user_id)
            if remaining < 1:
                raise HTTPException(409, "Cannot deactivate the last active admin")
        user.is_active = False
        audit.log_action(db, actor_id=actor.id, action="user_deactivate",
                         entity_type="User", entity_id=user.id)
        db.commit()
        db.refresh(user)
        return user

    @router.post("/{user_id}/reactivate", response_model=schemas.UserRead)
    def reactivate_user(
        user_id: UUID,
        db: Session = Depends(get_db),
        actor: models.User = Depends(require_role(models.UserRole.admin)),
    ):
        user = db.query(models.User).filter(models.User.id == user_id).first()
        if not user:
            raise HTTPException(404, "User not found")
        user.is_active = True
        audit.log_action(db, actor_id=actor.id, action="user_reactivate",
                         entity_type="User", entity_id=user.id)
        db.commit()
        db.refresh(user)
        return user
    ```

    ```python
    from datetime import date, datetime, timedelta, timezone

    # Anchor: Winter quarter 2026 first teaching day at UCSB (YYYY-01-06 is a common WQ start).
    # 11-week academic quarters per CLAUDE.md. If a Phase 5 helper already exists, replace
    # this with an import from that helper. TODO during execution: grep for existing anchor.
    QUARTER_ANCHOR: date = date(2026, 1, 6)
    WEEKS_PER_QUARTER: int = 11

    def _weeks_since_anchor(now: datetime) -> int:
        d = now.astimezone(timezone.utc).date() if now.tzinfo else now.date()
        return (d - QUARTER_ANCHOR).days // 7

    def quarter_index(now: datetime) -> int:
        return max(0, _weeks_since_anchor(now) // WEEKS_PER_QUARTER)

    def current_quarter_bounds(now: datetime) -> tuple[datetime, datetime]:
        idx = quarter_index(now)
        start_date = QUARTER_ANCHOR + timedelta(weeks=idx * WEEKS_PER_QUARTER)
        end_date = start_date + timedelta(weeks=WEEKS_PER_QUARTER)
        start = datetime.combine(start_date, datetime.min.time(), tzinfo=timezone.utc)
        end = datetime.combine(end_date, datetime.min.time(), tzinfo=timezone.utc)
        return (start, end)

    def previous_quarter_bounds(now: datetime) -> tuple[datetime, datetime]:
        idx = max(0, quarter_index(now) - 1)
        start_date = QUARTER_ANCHOR + timedelta(weeks=idx * WEEKS_PER_QUARTER)
        end_date = start_date + timedelta(weeks=WEEKS_PER_QUARTER)
        start = datetime.combine(start_date, datetime.min.time(), tzinfo=timezone.utc)
        end = datetime.combine(end_date, datetime.min.time(), tzinfo=timezone.utc)
        return (start, end)

    def quarter_progress(now: datetime) -> dict:
        start, _ = current_quarter_bounds(now)
        days_in = max(0, (now.astimezone(timezone.utc) - start).days)
        week = min(WEEKS_PER_QUARTER, 1 + days_in // 7)
        return {"week": week, "of": WEEKS_PER_QUARTER, "pct": round(week / WEEKS_PER_QUARTER, 2)}
    ```
